1 /**
2  * @file args.h Command-Line Argument Processing Support
3  *
4  * The command line processing consists of two phases. In the first phase, all expected arguments
5  * are registered with the args scanner. Finally, the arguments are scanned by a call to
6  * Scan().
7  *
8  * Example:
9  *
10  * @code
11  *
12  * void MyLooseArgHandler(void* context, const char* argValueString)
13  * {
14  * // Add argument string to list of things to be processed.
15  * ...
16  * }
17  *
18  * int main(int argc, char** argv)
19  * {
20  * bool isExtreme = false;
21  *
22  * args::AddOptionalFlag(&isExtreme, "x", "extreme", "Take it to the limit?");
23  * args::SetLooseArgHandler(MyLooseArgHandler, NULL);
24  *
25  * args::Scan();
26  *
27  * if (isExtreme)
28  * {
29  * // Do something extreme!
30  * ...
31  * }
32  * // Process list of things received as arguments.
33  * ...
34  * }
35  *
36  * @endcode
37  *
38  * <HR>
39  *
40  * Copyright (C) Sierra Wireless Inc.
41  */
