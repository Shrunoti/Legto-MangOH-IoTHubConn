2 /**
3  * @file pipeline.h API for creating pipes and pipelines of processes.
4  *
5  * The following code creates a pipeline "<c>cat foo | md5sum</c>" where the @c tar process reads
6  * its data from the parent process's standard input and the output from the @c md5sum process is
7  * delivered to an FD Monitor handler function called PipeOutputHandler() as it arrives.
8  *
9  * To make the completion callback work, the caller must monitor for SIGCHLD signals and call
10  * pipeline_CheckChildren() when it arrives.
11  *
12  * @code
13  *
14  * static void ChildSignalHandler(int sigNum)
15  * {
16  * pipeline_CheckChildren();
17  * }
18  *
19  * static void CatProcess(void* param)
20  * {
21  * const char* filePath = param;
22  *
23  * execl("cat", filePath, (char*)NULL);
24  * }
25  *
26  * static void Md5SumProcess(void)
27  * {
28  * execl("md5sum", "md5sum", (char*)NULL);
29  * }
30  *
31  * static void CompletionCallback
32  * (
33  * pipeline_Ref_t pipeline,
34  * int result // Exit status of the last process in the pipeline (see 'man waitpid').
35  * )
36  * {
37  * if (WIFEXITED(result) && WEXITSTATUS(EXIT_SUCCESS))
38  * {
39  * // Hooray!
40  * }
41  * else
42  * {
43  * // Boo! Killed by a signal or exited with non-zero exit code.
44  * }
45  * pipeline_Delete(pipeline);
46  * }
47  *
48  * static void RunPipeline()
49  * {
50  * pipeline_PipelineRef_t pipeline = pipeline_Create();
51  *
52  * pipeline_Append(pipeline, CatProcess, "foo");
53  *
54  * pipeline_Append(pipeline, Md5SumProcess, NULL);
55  *
56  * pipeline_SetInput(pipeline, STDIN_FILENO);
57  *
58  * fdMonitor_Create("my pipeline", pipeline_CreateOutputPipe(pipeline), PipeOutputHandler, POLLIN);
59  *
60  * pipeline_Start(pipeline, CompletionCallback);
61  * }
62  *
63  * COMPONENT_INIT
64  * {
65  * le_sig_Block(SIGCHLD);
66  * le_sig_SetEventHandler(SIGCHLD, ChildSignalHandler);
67  *
68  * RunPipeline();
69  * }
70  *
71  * @endcode
72  *
73  * <hr>
74  *
75  * Summary of available functions in this API:
76  *
77  * - pipeline_Create() creates a new pipeline.
78  *
79  * - pipeline_Delete() deletes a pipeline. Kills all remaining processes in the pipeline.
80  *
81  * - pipeline_Append() adds a process to the end of the pipeline.
82  *
83  * - pipeline_SetInput() provides a file descriptor for the pipeline to read its input from.
84  *
85  * - pipeline_CreateInputPipe() creates a pipe to be connected to the input of the first process
86  * in the pipeline. Returns the write end for the caller to use to write data into the pipe.
87  *
88  * - pipeline_SetOutput() provides a file descriptor for the pipeline to write its output to.
89  *
90  * - pipeline_CreateOutputPipe() creates a pipe to be connected to the output of the last process
91  * in the pipeline. Returns the read end for the caller to use to read data from the pipe.
92  *
93  * - pipeline_CreatePipe() can be used to create a pipe. It's a simple wrapper around pipe().
94  *
95  * - pipeline_Start() executes the processes in the pipeline. A completion callback function is
96  * provided, which will be called when the pipeline terminates.
97  *
98  * - pipeline_CheckChildren() must be called to tell the pipeline to check for the death
99  * of any of its children. This should be done in response to a SIGCHLD signal. It is
100  * recommended to use the @ref c_signals to detect SIGCHLD signals. DO NOT call any of this
101  * API's functions from inside an asynchronous signal handler. This API is NOT async safe.
102  *
103  * <b>Future Enhancements:</b>
104  *
105  * - pipeline_SetTimeout()
106  * - pipeline_Set/GetContextPtr()
107  * - pipeline_Signal() - Send a signal to all processes in the pipeline.
108  * - pipeline_Prepend()
109  *
110  * <hr>
111  *
112  * Copyright (C) Sierra Wireless Inc.
113  */
