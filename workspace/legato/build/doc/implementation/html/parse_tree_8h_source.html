2 /**
3  * @file parseTree.h
4  *
5  * Content_t is the base class for all file content items, including Token_t, and CompoundItem_t.
6  *
7  * The root of each parse tree is a DefFile_t. It has a pointer to the first token (Token_t) that
8  * was parsed from it. It also has a list of top-level sections (CompoundItem_t).
9  *
10  * Each Token_t keeps track of its exact text and the file, line, and column where it was found.
11  * As tokens are parsed from the file, they are linked together into a doubly-linked list.
12  * The DefFile_t has a pointer to the first Token_t in the file.
13  *
14  * Each CompoundItem_t has a type indicating what kind of item it is. It also has pointers to its
15  * first and last Token_t and a list of Content_t items that are inside it. If it is a simple
16  * section or named item with only a single name or number inside it, there will only be one
17  * Token_t pointer inside this list. If it is a more complex section or named item, there could
18  * be any number of Content_t objects in this content list, and it could even be empty if there's
19  * nothing but a pair of curly braces with nothing but whitespace or comments between them.
20  *
21  * <hr>
22  *
23  * Copyright (C) Sierra Wireless Inc.
24  */
