31  DOT, ///< '.'
32  STAR, ///< '*'
33  ARROW, ///< "->"
34  WHITESPACE, ///< Any combination of contiguous spaces, tabs, newlines and returns.
35  COMMENT, ///< A C/C++ style comment.
36  FILE_PERMISSIONS, ///< File permissions, in square brackets (e.g., "[rw]").
37  SERVER_IPC_OPTION, ///< Server-side IPC option, in square brackets (e.g., "[async]").
38  CLIENT_IPC_OPTION, ///< Client-side IPC option, in square brackets (e.g., "[types-only]").
39  ARG, ///< A command-line argument.
40  FILE_PATH, ///< A file system path.
41  FILE_NAME, ///< The name of a file or another name having the same constraints.
42  NAME, ///< Name safe to use as a program identifier in C-like languages.
43  DOTTED_NAME, ///< Name safe to use as a java package name.
44  GROUP_NAME, ///< Name safe to use as a user group name in Unix.
45  IPC_AGENT, ///< App or user name in a binding (e.g., "appName" or "<userName>").
46  INTEGER, ///< Integer number, possibly with a 'K' suffix.
47  SIGNED_INTEGER, ///< Like Integer, but supports both positive and negative values.
48  BOOLEAN, ///< Either "true" or "false".
49  FLOAT, ///< Standard C style floating point number.
50  STRING, ///< String value quoted with a ' or a ".
51  MD5_HASH, ///< MD5 cryptographic hash/checksum.
52  DIRECTIVE, ///< Preprocessor directive
53  };
